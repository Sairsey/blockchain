Moralis.Cloud.define("getNFT", async (request) => {
    const logger = Moralis.Cloud.getLogger();
    let NFTId = request.params.nftId;
    let hexId = parseInt(NFTId).toString(16);
    let paddedHex = ("00000000000000000000000000000000000000000000000000000000" + hexId).slice(-64)
    logger.info(paddedHex);
    return Moralis.Cloud.httpRequest({url: "https://ck5fx4ai9ebh.usemoralis.com" + paddedHex +  ".json"})
    .then(function(httpResponse){ 
        return httpResponse.text;
    })
})
